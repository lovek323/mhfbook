\question{Classify each of the following series as absolutely convergent, conditionally convergent, or divergent.

\begin{enumerate}
	\item
		\[\sum_{n=1}^\infty\frac{4^{2n}}{\left(-5\right)^n}\]
	\item
		\[\sum_{n=1}^\infty\sin\sqrt{\frac{1}{n}}\]
	\item
		\[\sum_{n=1}^\infty\left(-1\right)^{n-1}\frac{2^{3n}}{n^n}.\]
\end{enumerate}}{calculus/272-final-exams-series-help.html}{21 May 2005}

\index{series, Conditionally convergent}
\index{Conditionally convergent series}
\index{series, Absolutely convergent}
\index{Absolutely convergent series}
\index{series, Divergent}
\index{Divergent series}

\paragraph{Solution.}\partSolutionAuthor{hpe}

\begin{enumerate}
	\item
		We can use the \emph{ratio test} to determine whether this series converges.
	\item
		We can use the \emph{limit comparison test} to determine whether this series converges.
	\item
		We can use the \emph{root test} to determine whether this series converges (we could also have used the \emph{ratio test}, but the root test is more fun).
\end{enumerate}
